(

BEGIN

create FUNCTION [dbo].[getExpertCompanyPosition]
(	
	@ssn		VARCHAR(20)
 )
 RETURNS VARCHAR(56)
AS
BEGIN
 
  DECLARE @position VARCHAR(56)

  BEGIN
   SELECT @position = position
   FROM (
			select isNull(companyPosition, companyPositionname ) as position from expertpool
			where ssn = @ssn
   ) F
  END

  RETURN @position
 END
